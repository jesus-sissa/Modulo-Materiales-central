CODIGO PARA OCULTAR LAS COLUMNAS DEL LISTVIEW Y NO PUEDA VERLAS EL USUARIO NI MOVIENDO EL TAMAÑO DE LA COLUMNA



VARIABLE PRIVADA
Private _Ocultar As String = ""




PROPIEDAD
 Public Property Ocultar() As String
        Get
            Return _Ocultar
        End Get
        Set(ByVal value As String)
            _Ocultar = value
        End Set
    End Property



EVENTO MANEJADO
Private Sub cp_Listview_ColumnWidthChanging(ByVal sender As Object, ByVal e As System.Windows.Forms.ColumnWidthChangingEventArgs) Handles Me.ColumnWidthChanging
        If _Ocultar = "" Then Exit Sub
        Try
            Dim Omitir() As String = _Ocultar.Split(",")
            For Ilocal As Byte = 0 To Omitir.Length - 1
                If e.ColumnIndex = Omitir(Ilocal) Then
                    e.Cancel = True
                    e.NewWidth = 0
                    Exit Sub
                End If
            Next
        Catch
        End Try
    End Sub
